        Private Sub result()
            Dim op As String
            Dim iOp As Integer = 0

            If tb.Text.Contains("+") Then
                iOp = tb.Text.IndexOf("+")
            ElseIf tb.Text.Contains("-") Then
                iOp = tb.Text.IndexOf("-")
            ElseIf tb.Text.Contains("*") Then
                iOp = tb.Text.IndexOf("*")
            ElseIf tb.Text.Contains("/") Then
                iOp = tb.Text.IndexOf("/")
            Else
            End If

            op = tb.Text.Substring(iOp, 1)
            Dim op1 As Double = Convert.ToDouble(tb.Text.Substring(0, iOp))
            Dim op2 As Double = Convert.ToDouble(tb.Text.Substring(iOp + 1, tb.Text.Length - iOp - 1))

            If op = "+" Then
                tb.Text += "=" & (op1 + op2)
            ElseIf op = "-" Then
                tb.Text += "=" & (op1 - op2)
            ElseIf op = "*" Then
                tb.Text += "=" & (op1 * op2)
            Else
                tb.Text += "=" & (op1 / op2)
            End If
        End Sub
